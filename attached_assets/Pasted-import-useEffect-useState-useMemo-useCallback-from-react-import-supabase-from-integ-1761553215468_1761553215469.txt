import { useEffect, useState, useMemo, useCallback } from "react";
import { supabase } from "@/integrations/supabase/client";
import { MessageSquare, CheckCircle, XCircle, Activity, Search, ChevronDown } from "lucide-react";
import { useToast } from "@/hooks/use-toast";
import debounce from "lodash/debounce"; // IMPROVEMENT: For throttled refetches
import MessageCard from "@/components/MessageCard";
import StatsCard from "@/components/StatsCard";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input"; // IMPROVEMENT: Assume Shadcn Input; add if missing
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"; // IMPROVEMENT: For status filter
import { Skeleton } from "@/components/ui/skeleton"; // IMPROVEMENT: For loading skeletons

// IMPROVEMENT: Enhanced interface with stricter types (align with Supabase schema)
interface ForwardedMessage {
  id: string;
  whatsapp_message_id: string | null;
  sender_name: string;
  sender_contact: string;
  chat_name: string | null;
  message_content: string | null;
  media_type: string | null;
  media_url: string | null;
  timestamp: string;
  forwarded_at: string;
  discord_status: "pending" | "success" | "failed"; // Enum for safety
  error_message: string | null;
}

interface Stats {
  total: number;
  success: number;
  failed: number;
  pending: number;
}

const Index = () => {
  const [messages, setMessages] = useState<ForwardedMessage[]>([]);
  const [loading, setLoading] = useState(true);
  const [allMessages, setAllMessages] = useState<ForwardedMessage[]>([]); // IMPROVEMENT: Separate for filtering
  const [page, setPage] = useState(0); // IMPROVEMENT: Pagination state
  const [searchTerm, setSearchTerm] = useState(""); // IMPROVEMENT: Search state
  const [statusFilter, setStatusFilter] = useState("all"); // IMPROVEMENT: Filter state
  const [hasMore, setHasMore] = useState(true); // IMPROVEMENT: Pagination flag
  const [stats, setStats] = useState<Stats>({ total: 0, success: 0, failed: 0, pending: 0 });
  const { toast } = useToast();

  // IMPROVEMENT: Memoized stats calculation (avoids recalc on every render)
  const calculatedStats = useMemo(() => {
    const total = allMessages.length;
    return {
      total,
      success: allMessages.filter(m => m.discord_status === 'success').length,
      failed: allMessages.filter(m => m.discord_status === 'failed').length,
      pending: allMessages.filter(m => m.discord_status === 'pending').length,
    };
  }, [allMessages]);

  // IMPROVEMENT: Filtered messages with search + status
  const filteredMessages = useMemo(() => {
    return allMessages.filter(msg => {
      const matchesSearch = msg.sender_name.toLowerCase().includes(searchTerm.toLowerCase()) ||
        msg.sender_contact.includes(searchTerm) ||
        (msg.message_content && msg.message_content.toLowerCase().includes(searchTerm.toLowerCase()));
      const matchesStatus = statusFilter === "all" || msg.discord_status === statusFilter;
      return matchesSearch && matchesStatus;
    }).slice(0, (page + 1) * 50); // Paginate filtered results
  }, [allMessages, searchTerm, statusFilter, page]);

  // IMPROVEMENT: Paginated fetch with range
  const fetchMessages = useCallback(async (append = false) => {
    try {
      setLoading(true);
      const from = append ? page * 50 : 0;
      const to = from + 49; // 50 items
      const { data, error, count } = await supabase
        .from('forwarded_messages')
        .select('*', { count: 'exact' }) // IMPROVEMENT: Get total count for pagination
        .order('timestamp', { ascending: false })
        .range(from, to);

      if (error) throw error;

      if (append) {
        setAllMessages(prev => [...prev, ...(data || [])]);
      } else {
        setAllMessages(data || []);
        setPage(0);
      }
      setHasMore((data?.length || 0) === 50 && (count || 0) > to + 1); // IMPROVEMENT: Check for more pages
      setStats(calculatedStats); // Update stats from full data
    } catch (error: any) {
      toast({
        title: "Error fetching messages",
        description: error.message,
        variant: "destructive",
      });
      // IMPROVEMENT: Auto-retry once
      setTimeout(() => fetchMessages(append), 2000);
    } finally {
      setLoading(false);
    }
  }, [page, toast, calculatedStats]);

  // IMPROVEMENT: Debounced version for real-time
  const debouncedFetch = useMemo(() => debounce(fetchMessages, 500), [fetchMessages]);

  useEffect(() => {
    fetchMessages();

    // Subscribe to real-time (IMPROVEMENT: Use upsert for targeted updates)
    const channel = supabase
      .channel('forwarded_messages_changes')
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'forwarded_messages'
        },
        (payload) => {
          console.log('Real-time update:', payload);
          debouncedFetch(); // Throttled refetch
        }
      )
      .subscribe();

    return () => {
      supabase.removeChannel(channel);
      debouncedFetch.cancel(); // Cleanup debounce
    };
  }, [debouncedFetch]);

  // IMPROVEMENT: Load more handler
  const loadMore = () => {
    setPage(prev => prev + 1);
    fetchMessages(true);
  };

  // IMPROVEMENT: Clear filters
  const clearFilters = () => {
    setSearchTerm("");
    setStatusFilter("all");
    setPage(0);
    fetchMessages();
  };

  // Skeleton loader component (IMPROVEMENT: Inline for simplicity; extract if reused)
  const MessageSkeleton = () => (
    <div className="p-4 border rounded-lg">
      <Skeleton className="h-4 w-3/4 mb-2" />
      <Skeleton className="h-4 w-1/2" />
    </div>
  );

  return (
    <div className="min-h-screen bg-background">
      {/* Header - unchanged */}
      <header className="border-b bg-card">
        <div className="container mx-auto px-4 py-6">
          <div className="flex items-center justify-between">
            <div className="space-y-1">
              <h1 className="text-3xl font-bold flex items-center gap-2">
                <MessageSquare className="h-8 w-8 text-primary" />
                WhatsApp → Discord Bridge
              </h1>
              <p className="text-muted-foreground">
                Real-time message forwarding monitor
              </p>
            </div>
            <Badge variant="outline" className="gap-2">
              <Activity className="h-4 w-4 animate-pulse text-success" />
              Live
            </Badge>
          </div>
        </div>
      </header>

      {/* Stats Section - use calculatedStats */}
      <section className="container mx-auto px-4 py-8">
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
          <StatsCard
            title="Total Messages"
            value={calculatedStats.total}
            icon={MessageSquare}
            description="All forwarded messages"
          />
          <StatsCard
            title="Successful"
            value={calculatedStats.success}
            icon={CheckCircle}
            description="Successfully forwarded"
          />
          <StatsCard
            title="Failed"
            value={calculatedStats.failed}
            icon={XCircle}
            description="Failed to forward"
          />
          <StatsCard
            title="Pending"
            value={calculatedStats.pending}
            icon={Activity}
            description="Awaiting processing"
          />
        </div>
      </section>

      {/* Messages Section - enhanced with filters */}
      <section className="container mx-auto px-4 pb-12">
        <div className="flex flex-col sm:flex-row items-center justify-between mb-6 gap-4">
          <h2 className="text-2xl font-semibold">Recent Messages</h2>
          <div className="flex gap-2">
            <Input
              placeholder="Search by sender or content..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="w-64"
              leftIcon={<Search className="h-4 w-4" />} // Assume Input supports leftIcon
            />
            <Select value={statusFilter} onValueChange={setStatusFilter}>
              <SelectTrigger className="w-32">
                <SelectValue placeholder="All Statuses" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All</SelectItem>
                <SelectItem value="pending">Pending</SelectItem>
                <SelectItem value="success">Success</SelectItem>
                <SelectItem value="failed">Failed</SelectItem>
              </SelectContent>
            </Select>
            {(searchTerm || statusFilter !== "all") && (
              <Button onClick={clearFilters} variant="ghost" size="sm">
                Clear
              </Button>
            )}
            <Button onClick={() => fetchMessages()} variant="outline">
              Refresh
            </Button>
          </div>
        </div>

        {loading && page === 0 ? (
          <div className="grid grid-cols-1 gap-4">
            {Array.from({ length: 5 }).map((_, i) => <MessageSkeleton key={i} />)} {/* Initial load skeletons */}
          </div>
        ) : filteredMessages.length === 0 ? (
          <div className="text-center py-12 bg-card rounded-lg border">
            <MessageSquare className="h-16 w-16 text-muted-foreground mx-auto mb-4" aria-hidden="true" />
            <h3 className="text-xl font-semibold mb-2">No messages match your filters</h3>
            <p className="text-muted-foreground mb-4">
              {searchTerm || statusFilter !== "all" ? "Try adjusting your search." : "Messages will appear here once your WhatsApp bot starts forwarding."}
            </p>
            {searchTerm || statusFilter !== "all" && <Button onClick={clearFilters}>Clear Filters</Button>}
          </div>
        ) : (
          <>
            <div className="grid grid-cols-1 gap-4">
              {filteredMessages.map((message) => (
                <MessageCard
                  key={message.id}
                  senderName={message.sender_name}
                  senderContact={message.sender_contact}
                  chatName={message.chat_name || undefined}
                  messageContent={message.message_content || undefined}
                  mediaType={message.media_type || undefined}
                  mediaUrl={message.media_url || undefined} // IMPROVEMENT: Add alt={message.sender_name} in MessageCard for a11y
                  timestamp={message.timestamp}
                  discordStatus={message.discord_status}
                  errorMessage={message.error_message || undefined}
                />
              ))}
            </div>
            {hasMore && (
              <div className="text-center py-8">
                <Button onClick={loadMore} variant="outline" disabled={loading}>
                  {loading ? "Loading..." : "Load More"}
                  <ChevronDown className="ml-2 h-4 w-4" />
                </Button>
              </div>
            )}
          </>
        )}
      </section>
    </div>
  );
};

export default Index;
