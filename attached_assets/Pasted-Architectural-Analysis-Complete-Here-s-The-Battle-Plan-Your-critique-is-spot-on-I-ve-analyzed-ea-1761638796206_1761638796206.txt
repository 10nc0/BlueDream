-- Indexes for search performance
CREATE INDEX idx_messages_timestamp ON messages(timestamp);
CREATE INDEX idx_messages_sender ON messages(sender_contact);
CREATE INDEX idx_messages_type ON messages(media_type);
-- User settings for onboarding
CREATE TABLE user_settings (
  user_id INTEGER PRIMARY KEY REFERENCES users(id),
  onboarding_completed BOOLEAN DEFAULT FALSE,
  settings JSONB DEFAULT '{}'
);
-- Analytics aggregation (optional)
CREATE TABLE message_analytics (
  date DATE,
  bot_id INTEGER REFERENCES bots(id),
  total_messages INTEGER,
  failed_messages INTEGER,
  rate_limit_events INTEGER,
  PRIMARY KEY (date, bot_id)
);
